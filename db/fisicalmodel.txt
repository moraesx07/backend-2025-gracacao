Table users {
  id integer [primary key]
  username text
  email text
  password text
  role varchar [note: 'admin/parcial/comum']
  created_at timestamp
  updated_at timestamp

  indexes {
    (email, password) [name: 'login']
    (username) [name: 'name']
  }
}
